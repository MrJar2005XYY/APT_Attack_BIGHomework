# APT 攻击威胁情报的智能信息抽取系统 - 开发文档

## 1 项目概述

### 1.1 项目背景

随着网络攻击技术的不断演进，APT（高级持续性威胁）攻击已成为网络安全领域的主要挑战。APT 攻击具有高度隐蔽性、持续性和针对性，传统的安全防护手段难以有效应对。为了提升威胁情报的分析效率和准确性，本项目旨在构建一个智能化的 APT 攻击威胁情报信息抽取系统。

### 1.2 项目目标

本系统是一个基于 Web 的威胁情报智能分析平台，通过集成情报采集、自然语言处理、知识图谱和大模型分析等核心技术，实现对 APT 攻击情报的自动化采集、结构化抽取、可视化展示和智能分析。

### 1.3 系统架构

```
┌─────────────────────────────────────────────────────────────────┐
│                        前端展示层 (Vue3 + ECharts)              │
├─────────────────────────────────────────────────────────────────┤
│                        API 网关层 (FastAPI)                       │
├─────────────┬─────────────┬─────────────┬─────────────────────┤
│  情报采集模块 │  NLP 处理模块  │  大模型分析模块 │   知识图谱模块     │
├─────────────┴─────────────┴─────────────┴─────────────────────┤
│                        数据存储层 (MySQL + Neo4j)                │
└─────────────────────────────────────────────────────────────────┘
```

# 大模型接口统一抽象
class LLMProvider(ABC):
    """大模型 provider 抽象基类"""
    
    @abstractmethod
    def chat(self, messages: List[ChatMessage], **kwargs) -> str:
        """发送聊天请求"""
        pass
    
    @abstractmethod
    def get_model_name(self) -> str:
        """获取模型名称"""
        pass

class OpenAIProvider(LLMProvider):
    """OpenAI 官方 API 接入"""
    
    def __init__(self, config: OpenAIConfig):
        self.client = OpenAI(api_key=config.api_key, base_url=config.base_url)
        self.model = config.model
        
    def chat(self, messages: List[ChatMessage], **kwargs) -> str:
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[msg.dict() for msg in messages],
            temperature=kwargs.get("temperature", 0.7),
            max_tokens=kwargs.get("max_tokens", 2000)
        )
        return response.choices[0].message.content

class CustomOpenAIProvider(LLMProvider):
    """兼容 OpenAI 格式的第三方 API 接入（如 Azure OpenAI、Claude 等）"""
    
    def __init__(self, config: CustomOpenAIConfig):
        self.client = OpenAI(api_key=config.api_key, base_url=config.base_url)
        self.model = config.model
        self.provider_name = config.provider_name
        
    def chat(self, messages: List[ChatMessage], **kwargs) -> str:
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[msg.dict() for msg in messages],
            temperature=kwargs.get("temperature", 0.7),
            max_tokens=kwargs.get("max_tokens", 2000)
        )
        return response.choices[0].message.content
    
    def get_model_name(self) -> str:
        return f"{self.provider_name}/{self.model}"

#### 2.4.3 技术规范

```python
from neo4j import GraphDatabase

class KnowledgeGraph:
    """知识图谱管理"""
    
    def __init__(self, config: Neo4jConfig):
        self.driver = GraphDatabase.driver(
            config.uri,
            auth=(config.username, config.password)
        )
    
    def create_entity(self, entity: ThreatEntity):
        """创建实体节点"""
        query = """
        MERGE (e:ThreatEntity {name: $name})
        SET e.type = $type,
            e.aliases = $aliases,
            e.description = $description,
            e.confidence = $confidence
        """
        with self.driver.session() as session:
            session.run(query, **entity.dict())
    
    def create_relation(self, relation: ThreatRelation):
        """创建关系"""
        query = """
        MATCH (s:ThreatEntity {name: $source_entity})
        MATCH (t:ThreatEntity {name: $target_entity})
        MERGE (s)-[r:RELATES_TO {type: $relation_type}]->(t)
        SET r.evidence = $evidence,
            r.confidence = $confidence,
            r.source = $source
        """
        with self.driver.session() as session:
            session.run(query, **relation.dict())
    
    def query_apt_relations(self, apt_name: str) -> dict:
        """查询 APT 组织的关联信息"""
        query = """
        MATCH (apt:ThreatEntity {name: $apt_name})-[r]->(target)
        RETURN type(r) as relation_type, target.name as target_name, 
               target.type as target_type
        """
        with self.driver.session() as session:
            result = session.run(query, apt_name=apt_name)
            return [dict(record) for record in result]
    
    def find_similar_attack_patterns(self, techniques: List[str]) -> List[dict]:
        """查找相似的攻击模式"""
        query = """
        MATCH (apt:ThreatEntity)-[:USES_TECHNIQUE]->(t:ThreatEntity)
        WHERE t.name IN $techniques
        WITH apt, collect(t.name) as used_techniques
        WHERE size([x IN $techniques WHERE x IN used_techniques]) >= size($techniques) * 0.5
        RETURN apt.name as apt_name, used_techniques, 
               size([x IN $techniques WHERE x IN used_techniques]) as match_count
        ORDER BY match_count DESC
        LIMIT 10
        """
        with self.driver.session() as session:
            result = session.run(query, techniques=techniques)
            return [dict(record) for record in result]
```
